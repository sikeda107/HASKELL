superimposeChar :: Char -> Char -> Char
superimposeChar '.' '.' = '.'
superimposeChar _   _   = '#'

superimposeLine :: [Char] -> [Char] -> [Char]
superimposeLine a b = [ superimposeChar a b | (a,b) <- zip a b ]

superimpose :: Picture -> Picture -> Picture
superimpose a b = [ superimposeLine a b | (a,b) <- zip a b  ]
